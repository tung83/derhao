jQuery asSwitch Plugin Demos

skin


<input class="sw-3" type="checkbox" checked="checked" />
<script>                   
    $(document).ready(function() {
        $('.sw-3').asSwitch({
            skin: 'skin-3'
        });   
                      
    });
</script>

only drag


<input class="sw-drag" type="checkbox" checked="checked" />
<script>                   
    $(document).ready(function() {
        $('.sw-drag').asSwitch({
            dragable: true,
            clickable: false,
            skin: 'skin-6'
        });   
                      
    });
</script>
